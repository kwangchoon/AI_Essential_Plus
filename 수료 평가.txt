examples = [
    ("Q: reset 방법?", "A: 설정 > 일반 > 초기화"),
    ("Q: 와이파이 연결?", "A: 설정 > 네트워크"),

Q4.
객관식(단일선택)|배점 : 10점
다음 코드가 출력하는 리스트에서 인덱스 2(세 번째 원소) 는 무엇인가?

from collections import deque

def a2a_run():
    q = deque([("planner", "요구사항 분석"), ("executor", "도구 실행")])
    log = []
    while q:
        agent, msg = q.popleft()
        log.append(f"{agent}:{msg}")

        if agent == "planner":
            q.append(("executor", "검색 수행"))
        elif agent == "executor" and "검색" in msg:
            q.append(("writer", "응답 생성"))
    return log

print(a2a_run())

2

3

4

query = "hdmi 입력"
docs = {
    "D1": ("hdmi 설정 메뉴", 0.62),
    "D2": ("wifi 연결 방법", 0.70),
    "D3": ("hdmi 입력 전환", 0.60),
}

top_k = 2
cands = sorted(docs.items(), key=lambda x: x[1][1], reverse=True)[:top_k]

q_tokens = set(query.split())
def overlap(text: str):
    return len(q_tokens & set(text.split()))

reranked = sorted(cands, key=lambda x: overlap(x[1][0]), reverse=True)
print([d for d, _ in cands], "->", [d for d, _ in reranked][0])

Q8.
객관식(단일선택)|배점 : 10점
다음 코드에서 top으로 선택되는 문서 ID는?

import math

docs = {
    "A": [1, 0],
    "B": [0, 1],
    "C": [1, 1],
}
q = [1, 0.2]

def cos(u, v):
    dot = sum(a*b for a, b in zip(u, v))
    nu = math.sqrt(sum(a*a for a in u))
    nv = math.sqrt(sum(b*b for b in v))
    return dot / (nu * nv)

scores = {k: round(cos(vec, q), 3) for k, vec in docs.items()}
top = max(scores, key=scores.get)
print(scores, top)
A.
1
B
2
동률이라 결정 불가
3
A
4
C

Q9.
객관식(단일선택)|배점 : 10점
다음 코드의 출력으로 가장 알맞은 것은?

TOOLS = {
    "calc": lambda x: eval(x),
    "search": lambda q: f"검색:{q.split()[0]}",
}

def agent(user_msg: str):
    if user_msg.startswith("계산:"):
        tool, arg = "calc", user_msg.replace("계산:", "").strip()
    elif ("찾아" in user_msg) or ("검색" in user_msg):
        tool, arg = "search", user_msg.replace("검색", "").strip()
    else:
        return "LLM_ONLY"
    return f"{tool} -> {TOOLS[tool](arg)}"

print(agent("계산: 2**3 + 1"))
A.
1
calc -> 8
2
LLM_ONLY
3
calc -> 9
4
search -> 검색:2**3
